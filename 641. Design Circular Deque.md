# 641. Design Circular Deque

链接：https://leetcode-cn.com/problems/design-circular-deque/

## 数组循环队列解法

```go
type MyCircularDeque struct {
    front, rear int
    numbers []int
    full bool
}


/** Initialize your data structure here. Set the size of the deque to be k. */
func Constructor(k int) MyCircularDeque {
    return MyCircularDeque{
        numbers: make([]int, k),
    }
}


/** Adds an item at the front of Deque. Return true if the operation is successful. */
func (this *MyCircularDeque) InsertFront(value int) bool {
    if this.IsFull() { return false }
    this.front = (this.front+cap(this.numbers)-1) % cap(this.numbers)
    this.numbers[this.front] = value
    this.full = this.front == this.rear
    return true
}


/** Adds an item at the rear of Deque. Return true if the operation is successful. */
func (this *MyCircularDeque) InsertLast(value int) bool {
    if this.IsFull() { return false }
    this.numbers[this.rear] = value
    this.rear = (this.rear+1) % cap(this.numbers)
    this.full = this.rear == this.front
    return true
}


/** Deletes an item from the front of Deque. Return true if the operation is successful. */
func (this *MyCircularDeque) DeleteFront() bool {
    if this.IsEmpty() { return false }
    this.front = (this.front+1) % cap(this.numbers)
    this.full = false
    return true
}


/** Deletes an item from the rear of Deque. Return true if the operation is successful. */
func (this *MyCircularDeque) DeleteLast() bool {
    if this.IsEmpty() { return false }
    this.rear = (this.rear+cap(this.numbers)-1) % cap(this.numbers)
    this.full = false
    return true
}


/** Get the front item from the deque. */
func (this *MyCircularDeque) GetFront() int {
    if this.IsEmpty() { return -1 }
    return this.numbers[this.front]
}


/** Get the last item from the deque. */
func (this *MyCircularDeque) GetRear() int {
    if this.IsEmpty() { return -1 }
    rear := (this.rear+cap(this.numbers)-1) % cap(this.numbers)
    return this.numbers[rear]
}


/** Checks whether the circular deque is empty or not. */
func (this *MyCircularDeque) IsEmpty() bool {
    return this.front == this.rear && !this.full
}


/** Checks whether the circular deque is full or not. */
func (this *MyCircularDeque) IsFull() bool {
    return this.full
}

 * param_4 := obj.DeleteLast();
 * param_5 := obj.GetFront();
 * param_6 := obj.GetRear();
 * param_7 := obj.IsEmpty();
 * param_8 := obj.IsFull();
 */
```
